/// MIK-6558.AC.2 AC.2: Read-only inventory and evidence APIs ship before any mutation API is enabled: GET endpoints expose server/tool inventory, runtime health, TrustCard/eval summaries, approvals, and audit evidence, while POST/PATCH/DELETE control-plane routes return disabled/not-implemented until the reconciliation layer is present. CHECK: `cargo test --all-features control_plane_read_only_slice_blocks_mutations` exits 0 (expected: read endpoints return 200 with typed JSON and mutation endpoints return 403 or 501)
#[cfg(feature = "enterprise")]
#[tokio::test]
async fn ac_2_mik_6558_ac_2_ac_2_read_only_inventory_and_evid() {
    use axum::{body::Body, http::{Request, StatusCode}};
    use std::sync::Arc;
    use tower::ServiceExt;
    use control_plane::api::{ControlPlaneApiState, control_plane_router};
    use control_plane::license::{LicenseGate, LicenseTier};
    use control_plane::rbac::RbacEngine;
    use control_plane::storage::EmbeddedControlPlaneStore;

    let state = Arc::new(ControlPlaneApiState {
        store: Arc::new(EmbeddedControlPlaneStore::new()),
        rbac: RbacEngine::new(),
        license: LicenseGate::new(LicenseTier::Enterprise),
    });
    let app = control_plane_router(state);

    // GET endpoints return 200 (or 404 for missing resources)
    let read_endpoints = [
        "/ui/api/control-plane/servers",
        "/ui/api/control-plane/tools",
        "/ui/api/control-plane/health",
        "/ui/api/control-plane/trust-cards",
        "/ui/api/control-plane/grants",
        "/ui/api/control-plane/grants/test-id",
        "/ui/api/control-plane/policies",
        "/ui/api/control-plane/policies/test-id",
        "/ui/api/control-plane/approvals",
        "/ui/api/control-plane/approvals/test-id",
        "/ui/api/control-plane/audit",
        "/ui/api/control-plane/users",
        "/ui/api/control-plane/groups",
    ];
    for endpoint in &read_endpoints {
        let req = Request::builder()
            .method("GET").uri(*endpoint)
            .header("X-Control-Plane-Role", "admin")
            .body(Body::empty()).unwrap();
        let resp = app.clone().oneshot(req).await.unwrap();
        let s = resp.status();
        assert!(s == StatusCode::OK || s == StatusCode::NOT_FOUND,
            "GET {endpoint}: expected 200/404, got {s}");
    }

    // POST/PATCH/DELETE mutation endpoints return 501 or 403
    let mutation_endpoints = [
        ("POST", "/ui/api/control-plane/grants"),
        ("PATCH", "/ui/api/control-plane/grants/test-id"),
        ("DELETE", "/ui/api/control-plane/grants/test-id"),
        ("POST", "/ui/api/control-plane/policies"),
        ("PATCH", "/ui/api/control-plane/policies/test-id"),
        ("DELETE", "/ui/api/control-plane/policies/test-id"),
        ("POST", "/ui/api/control-plane/approvals"),
        ("POST", "/ui/api/control-plane/approvals/test-id/approve"),
        ("POST", "/ui/api/control-plane/approvals/test-id/reject"),
        ("POST", "/ui/api/control-plane/export"),
    ];
    for (method, endpoint) in &mutation_endpoints {
        let req = Request::builder()
            .method(*method).uri(*endpoint)
            .header("X-Control-Plane-Role", "admin")
            .body(Body::empty()).unwrap();
        let resp = app.clone().oneshot(req).await.unwrap();
        let s = resp.status();
        assert!(s == StatusCode::NOT_IMPLEMENTED || s == StatusCode::FORBIDDEN,
            "{method} {endpoint}: expected 501/403, got {s}");
    }
}

/// MIK-6558.AC.3 AC.3: RBAC enforces admin, security_reviewer, developer, and auditor behavior: non-admin users cannot directly mutate applied grants/policies, security reviewers can approve/reject but not bypass reconciliation, developers can request but not approve their own requests, and auditors remain read-only. CHECK: `cargo test --all-features control_plane_rbac_matrix` exits 0 (expected: role/action matrix matches the four-role contract)
#[cfg(feature = "enterprise")]
#[test]
fn ac_3_mik_6558_ac_3_ac_3_rbac_enforces_admin_securit() {
    use control_plane::rbac::{RbacEngine, Role, Action};

    let engine = RbacEngine::new();

    // Admin: can do everything
    assert!(engine.can(&Role::Admin, &Action::CreateGrant));
    assert!(engine.can(&Role::Admin, &Action::ManageUsers));
    assert!(engine.can(&Role::Admin, &Action::ApproveApproval));

    // SecurityReviewer: approve/reject but NOT mutate directly
    assert!(engine.can(&Role::SecurityReviewer, &Action::ApproveApproval));
    assert!(engine.can(&Role::SecurityReviewer, &Action::RejectApproval));
    assert!(!engine.can(&Role::SecurityReviewer, &Action::CreateGrant));
    assert!(!engine.can(&Role::SecurityReviewer, &Action::UpdateGrant));
    assert!(!engine.can(&Role::SecurityReviewer, &Action::CreatePolicy));
    assert!(!engine.can(&Role::SecurityReviewer, &Action::ManageUsers));

    // Developer: request (create approval) but NOT approve
    assert!(engine.can(&Role::Developer, &Action::CreateApproval));
    assert!(!engine.can(&Role::Developer, &Action::ApproveApproval));
    assert!(!engine.can(&Role::Developer, &Action::RejectApproval));
    assert!(!engine.can(&Role::Developer, &Action::CreateGrant));
    assert!(!engine.can(&Role::Developer, &Action::ManageUsers));

    // Auditor: read-only
    assert!(engine.can(&Role::Auditor, &Action::ListGrants));
    assert!(engine.can(&Role::Auditor, &Action::ListAuditEvidence));
    assert!(engine.can(&Role::Auditor, &Action::ExportEvidence));
    assert!(!engine.can(&Role::Auditor, &Action::CreateGrant));
    assert!(!engine.can(&Role::Auditor, &Action::UpdateGrant));
    assert!(!engine.can(&Role::Auditor, &Action::ApproveApproval));
    assert!(!engine.can(&Role::Auditor, &Action::RejectApproval));
    assert!(!engine.can(&Role::Auditor, &Action::ManageUsers));

    // Self-approval: only Admin
    assert!(engine.can_approve_own_request(&Role::Admin));
    assert!(!engine.can_approve_own_request(&Role::SecurityReviewer));
    assert!(!engine.can_approve_own_request(&Role::Developer));
    assert!(!engine.can_approve_own_request(&Role::Auditor));

    // Developer cannot approve own
    assert!(!engine.can_approve_own_request(&Role::Developer));
    assert!(!engine.can(&Role::Developer, &Action::ApproveApproval));

    // Security reviewer cannot bypass reconciliation (no direct mutations)
    assert!(!engine.can(&Role::SecurityReviewer, &Action::CreateGrant));
    assert!(!engine.can(&Role::SecurityReviewer, &Action::UpdateGrant));
    assert!(!engine.can(&Role::SecurityReviewer, &Action::DeleteGrant));
    assert!(!engine.can(&Role::SecurityReviewer, &Action::CreatePolicy));
    assert!(!engine.can(&Role::SecurityReviewer, &Action::UpdatePolicy));
    assert!(!engine.can(&Role::SecurityReviewer, &Action::DeletePolicy));
}

/// MIK-6558.AC.4 AC.4: Grant and policy mutations require approval, emit durable audit evidence, record previous/applied/rollback states, and can be rolled back through the same reconciler without direct config/database edits. CHECK: `cargo test --all-features control_plane_grant_policy_reconciliation_audits_and_rolls_back` exits 0 (expected: mutation, approval, apply, rollback, and audit assertions all pass)
#[cfg(feature = "enterprise")]
#[tokio::test]
async fn ac_4_mik_6558_ac_4_ac_4_grant_and_policy_mutations_r() {
    use std::sync::Arc;
    use control_plane::domain::{IdentityGrant, GrantState, PolicyBinding, ApprovalStatus};
    use control_plane::license::{LicenseGate, LicenseTier};
    use control_plane::rbac::{RbacEngine, Role};
    use control_plane::reconciler::ControlPlaneReconciler;
    use control_plane::storage::EmbeddedControlPlaneStore;
    use chrono::Utc;

    let store = Arc::new(EmbeddedControlPlaneStore::new());
    let rbac = RbacEngine::new();
    let license = LicenseGate::new(LicenseTier::Enterprise);
    let reconciler = ControlPlaneReconciler::new(store, rbac, license);

    // Grant lifecycle: request → approve → rollback
    let grant = IdentityGrant {
        id: "g-test".into(), name: "Test Grant".into(), description: Some("desc".into()),
        subject: "user:alice".into(), resource: "tool:search".into(),
        state: GrantState::Applied, previous_state: None, rollback_state: None,
        created_by: String::new(), created_at: Utc::now(), updated_at: Utc::now(),
        version: 0, metadata: serde_json::json!({}),
    };

    // Admin requests creation
    let approval = reconciler.request_create_grant(grant, "admin", &Role::Admin, Some("t-1"))
        .await.expect("request create grant");
    assert_eq!(approval.status, ApprovalStatus::Pending);

    // Security reviewer approves
    let applied = reconciler.approve_grant(&approval.id, "sec-reviewer", &Role::SecurityReviewer,
        Some("LGTM".into()), Some("t-2")).await.expect("approve grant");
    assert_eq!(applied.state, GrantState::Applied);

    // Rollback
    let rolled = reconciler.rollback_grant(&applied.id, "admin", &Role::Admin,
        Some("not needed".into()), Some("t-3")).await.expect("rollback grant");
    assert_eq!(rolled.state, GrantState::RolledBack);
    assert!(rolled.rollback_state.is_some());

    // Developer cannot approve their own
    let grant2 = IdentityGrant {
        id: "g-dev".into(), name: "Dev Grant".into(), description: None,
        subject: "user:bob".into(), resource: "tool:read".into(),
        state: GrantState::Applied, previous_state: None, rollback_state: None,
        created_by: String::new(), created_at: Utc::now(), updated_at: Utc::now(),
        version: 0, metadata: serde_json::json!({}),
    };
    let dev_approval = reconciler.request_create_grant(grant2, "dev", &Role::Developer, None)
        .await;
    // Developer does NOT have CreateGrant RBAC
    assert!(dev_approval.is_err());

    // Policy lifecycle
    let binding = PolicyBinding {
        id: "pb-test".into(), name: "Test Policy".into(), policy_type: "allowlist".into(),
        target: "srv:brave".into(), rules: serde_json::json!({"allow":["*"]}),
        priority: 10, enabled: true, created_by: String::new(),
        created_at: Utc::now(), updated_at: Utc::now(), version: 0,
    };
    let pb_approval = reconciler.request_create_policy(binding, "admin", &Role::Admin, Some("t-4"))
        .await.expect("request create policy");
    let applied_pb = reconciler.approve_policy(&pb_approval.id, "sec-reviewer", &Role::SecurityReviewer,
        None, Some("t-5")).await.expect("approve policy");
    assert_eq!(applied_pb.name, "Test Policy");

    let rolled_pb = reconciler.rollback_policy(&applied_pb.id, "admin", &Role::Admin,
        Some("deprecated".into()), Some("t-6")).await.expect("rollback policy");
    assert!(!rolled_pb.enabled);

    // Audit evidence was emitted
    let audit = reconciler.store.list_audit_evidence(None, None).await.unwrap();
    assert!(audit.len() >= 6, "Expected >=6 audit records, got {}", audit.len());
    for ev in &audit {
        assert!(!ev.id.is_empty());
        assert!(!ev.event_type.is_empty());
        assert!(!ev.actor.is_empty());
        assert!(!ev.decision.is_empty());
    }
}
