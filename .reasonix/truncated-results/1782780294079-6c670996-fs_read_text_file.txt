/// MIK-6555.AC.1 AC.1: `Backend::start` delegates stdio backend lifecycle to a
/// `RuntimeProvider` abstraction instead of constructing `StdioTransport`
/// directly, and the trait covers start, stop, health/readiness, logs, resource
/// policy, secrets/env materialization, mounts, network policy, and audit
/// evidence.
///
/// CHECK: `cargo test --lib runtime_provider_trait_covers_lifecycle_policy_and_evidence`
/// exits 0 (expected: test asserts trait-backed fake provider is used by
/// backend startup and exposes all required operation types).
#[test]
fn ac_1_mik_6555_ac_1_ac_1_backend_start_delegates_s() {
    // ── Provider factory creates correct types ───────────────────────────
    let local = create_provider(&RuntimeConfig::local_compat())
        .expect("local_compat provider should be creatable");
    assert_eq!(local.provider_id(), "local_compat");

    let docker = create_provider(&RuntimeConfig::docker_restricted())
        .expect("docker provider should be creatable");
    assert_eq!(docker.provider_id(), "docker");

    let mut podman_cfg = RuntimeConfig::docker_restricted();
    podman_cfg.provider = "podman".to_string();
    let podman = create_provider(&podman_cfg)
        .expect("podman provider should be creatable");
    assert_eq!(podman.provider_id(), "podman");

    // ── Unknown provider returns error ──────────────────────────────────
    let mut unknown = RuntimeConfig::local_compat();
    unknown.provider = "nonexistent".to_string();
    assert!(create_provider(&unknown).is_err(), "unknown provider must error");

    // ── Trait covers resource policy validation ─────────────────────────
    let local_config = RuntimeConfig::local_compat();
    let verdict = local.validate_policy(&local_config);
    assert!(verdict.is_allowed(), "default local_compat policy must be allowed");

    // ── Trait covers egress validation ──────────────────────────────────
    let mut egress_config = RuntimeConfig::local_compat();
    egress_config.egress.deny_default = true;
    let egress_verdict = local.validate_policy(&egress_config);
    assert!(!egress_verdict.is_allowed(),
        "local_compat must deny egress.deny_default");

    // ── Trait covers mount validation ───────────────────────────────────
    let mount_entry = MountEntry {
        host: "/var/run/docker.sock".to_string(),
        container: "/var/run/docker.sock".to_string(),
        writable: false,
    };
    let mount_policy = MountPolicy::default_restricted();
    let mount_verdict = validate_mount(&mount_entry, &mount_policy);
    assert!(!mount_verdict.is_allowed(), "docker socket mount must be denied");

    // ── Trait exposes audit evidence ────────────────────────────────────
    let events = local.audit_selection("test-backend", &RuntimeConfig::local_compat());
    assert!(!events.is_empty(), "audit events must be emitted");
    assert_eq!(events[0].provider, "local_compat");
    assert_eq!(events[0].backend, "test-backend");

    // ── RuntimeConfig covers all required policy fields ─────────────────
    let mut cfg = RuntimeConfig::docker_restricted();
    cfg.resources = ResourcePolicy { cpu: 1.0, memory: "512MiB".to_string() };
    cfg.mounts = MountPolicy {
        mounts: vec![MountEntry {
            host: "/tmp/data".to_string(),
            container: "/data".to_string(),
            writable: false,
        }],
        allow_writable: false,
    };
    cfg.egress = EgressPolicy { deny_default: true, allowlist: vec!["10.0.0.0/8".to_string()] };
    cfg.env_policy = EnvPolicy { allowlist: vec!["PATH".to_string()], inherit_env: false };
    cfg.secrets = SecretPolicy {
        env_secrets: {
            let mut m = HashMap::new();
            m.insert("API_KEY".to_string(), "env:SECRET".to_string());
            m
        },
        file_secrets: HashMap::new(),
    };
    cfg.identity = IdentityPolicy { user: Some("1000".to_string()), group: None };
    cfg.timeouts = TimeoutPolicy { start_secs: 60, stop_secs: 10, idle_timeout_secs: 300 };
    cfg.log_policy = LogPolicy { capture: true, max_lines: 500 };
    assert_eq!(cfg.provider, "docker");
}

// AC.3 — BackendConfig parses and serializes runtime policy
// ═══════════════════════════════════════════════════════════════════════════════

/// MIK-6555.AC.3 AC.3: `BackendConfig` parses and serializes `runtime.provider`
/// plus canonical policy fields for resources, filesystem mounts, egress,
/// env allowlist, secrets, identity, timeouts, and log capture, with omitted
/// `runtime` defaulting to `local_compat`.
///
/// CHECK: `cargo test --lib runtime_policy_config_defaults_and_roundtrip` exits 0
/// (expected: YAML without runtime becomes `local_compat`; YAML with docker
/// policy round-trips without losing fields).
#[test]
fn ac_3_mik_6555_ac_3_ac_3_backendconfig_parses_and_s() {
    // ── YAML without runtime becomes local_compat ────────────────────────
    let yaml_no_runtime = r#"
backends:
  my-backend:
    command: "./my-server"
"#;
    let config: mcp_gateway::config::Config =
        serde_yaml::from_str(yaml_no_runtime).expect("parse YAML");
    let backend = config.backends.get("my-backend").unwrap();
    assert_eq!(backend.runtime.provider, "local_compat",
        "omitted runtime must default to local_compat");

    // ── YAML with docker policy round-trips without losing fields ───────
    let yaml_docker = r#"
backends:
  my-backend:
    command: "./my-server"
    runtime:
      provider: docker
      resources:
        cpu: 1.0
        memory: "512MiB"
      mounts:
        allow_writable: true
        mounts:
          - host: /tmp/data
            container: /data
            writable: true
      egress:
        deny_default: true
        allowlist:
          - "10.0.0.0/8"
      env_policy:
        inherit_env: false
        allowlist:
          - PATH
      secrets:
        env_secrets:
          API_KEY: "env:MCP_SECRET"
      identity:
        user: "1000"
      timeouts:
        start_secs: 60
      log_policy:
        capture: true
        max_lines: 500
"#;
    let config: mcp_gateway::config::Config =
        serde_yaml::from_str(yaml_docker).expect("parse docker YAML");
    let backend1 = config.backends.get("my-backend").unwrap();
    assert_eq!(backend1.runtime.provider, "docker");

    // Round-trip through serialization
    let roundtripped_yaml = serde_yaml::to_string(&config)
        .expect("serialize to YAML");
    let config2: mcp_gateway::config::Config =
        serde_yaml::from_str(&roundtripped_yaml).expect("parse round-tripped YAML");
    let backend2 = config2.backends.get("my-backend").unwrap();

    // Verify all fields preserved
    assert_eq!(backend2.runtime.provider, "docker");
    assert!((backend2.runtime.resources.cpu - 1.0f64).abs() < f64::EPSILON);
    assert_eq!(backend2.runtime.resources.memory, "512MiB");
    assert!(backend2.runtime.mounts.allow_writable);
    assert_eq!(backend2.runtime.mounts.mounts.len(), 1);
    assert_eq!(backend2.runtime.mounts.mounts[0].host, "/tmp/data");
    assert_eq!(backend2.runtime.mounts.mounts[0].container, "/data");
    assert!(backend2.runtime.mounts.mounts[0].writable);
    assert!(backend2.runtime.egress.deny_default);
    assert_eq!(backend2.runtime.egress.allowlist.len(), 1);
    assert!(!backend2.runtime.env_policy.inherit_env);
    assert_eq!(backend2.runtime.env_policy.allowlist.len(), 1);
    assert_eq!(backend2.runtime.secrets.env_secrets.len(), 1);
    assert_eq!(backend2.runtime.identity.user.as_deref(), Some("1000"));
    assert_eq!(backend2.runtime.timeouts.start_secs, 60);
    assert!(backend2.runtime.log_policy.capture);
    assert_eq!(backend2.runtime.log_policy.max_lines, 500);

    // ── BackendConfig default runtime is local_compat ────────────────────
    let default_cfg = BackendConfig::default();
    assert_eq!(default_cfg.runtime.provider, "local_compat");
}

/// MIK-6555.AC.4 AC.4: Docker/Podman provider starts a committed fixture stdio
/// MCP server with restricted defaults: no host network, read-only root
/// filesystem when no writable mount is declared, explicit env allowlist only,
/// read-only bind mounts by default, resource flags for configured CPU/memory,
/// deterministic labels, and no privileged/seccomp-unconfined/Docker-socket
/// mounts.
///
/// CHECK: `cargo test --test runtime_provider_docker
/// docker_podman_fixture_starts_with_restricted_defaults -- --ignored` exits 0
/// (expected: fixture responds to `initialize` and captured runtime args
/// include `--network none`, `--read-only`, explicit `--env`, CPU/memory
/// flags, and no forbidden flags).
///
/// NOTE: The actual —ignored integration test lives in
/// `tests/runtime_provider_docker.rs`. This test verifies the policy-level
/// defaults without requiring a running Docker daemon.
#[test]
fn ac_4_mik_6555_ac_4_ac_4_docker_podman_provider_start() {
    // ── Verify restricted defaults in policy ─────────────────────────────
    let config = RuntimeConfig::docker_restricted();
    assert_eq!(config.provider, "docker");
    assert!(config.egress.deny_default, "egress must deny by default");
    assert!(config.egress.allowlist.is_empty(), "no allowlist entries by default");
    assert!(config.mounts.mounts.is_empty(), "no mounts by default");
    assert!(!config.mounts.allow_writable, "writable must be disabled by default");

    // ── Docker provider accepts restricted config ───────────────────────
    let provider = DockerProvider::new("docker");
    let verdict = provider.validate_policy(&config);
    assert!(verdict.is_allowed(), "restricted config must pass validation");

    // ── Podman provider also accepts restricted config ──────────────────
    let podman_cfg = RuntimeConfig::podman_restricted();
    assert_eq!(podman_cfg.provider, "podman");
    let podman_provider = DockerProvider::new("podman");
    let podman_verdict = podman_provider.validate_policy(&podman_cfg);
    assert!(podman_verdict.is_allowed(), "podman restricted config must pass");

    // ── Docker socket mount is denied at policy level ───────────────────
    let mut config_with_sock = RuntimeConfig::docker_restricted();
    config_with_sock.mounts.mounts.push(MountEntry {
        host: "/var/run/docker.sock".to_string(),
        container: "/var/run/docker.sock".to_string(),
        writable: false,
    });
    let sock_verdict = provider.validate_policy(&config_with_sock);
    assert!(!sock_verdict.is_allowed(), "docker socket mount must be denied");

    // ── Host network is denied via --network=host flag (enforced at spawn) ─
    // Host network is not a RuntimeConfig field; it's enforced during
    // container spawn by detecting forbidden flags. No policy field maps
    // to --network=host directly — DockerProvider defaults to --network none.

    // ── Resource flags: CPU and memory configured ──────────���────────────
    let mut config_with_resources = RuntimeConfig::docker_restricted();
    config_with_resources.resources = ResourcePolicy {
        cpu: 1.0,
        memory: "512MiB".to_string(),
    };
    let res_verdict = provider.validate_policy(&config_with_resources);
    assert!(res_verdict.is_allowed(), "valid resource policy must pass");

    // ── Audit events verify provider selection ──────────────────────────
    let events = provider.audit_selection("docker-backend", &config);
    assert_eq!(events[0].provider, "docker");
    assert_eq!(events[0].backend, "docker-backend");
}

// AC.5 — Denied network and mount policies fail closed before spawn
// ═════════��═════════════════════════════════════════════════════════════════════

/// MIK-6555.AC.5 AC.5: Denied network and mount policies fail closed before
/// process/container start, including host network, Docker socket mount, `/`,
/// `/etc`, `/proc`, `/sys`, `/dev`, `/var/run`, relative paths, path
/// traversal, writable mount without explicit `writable=true`, and egress
/// allowlist that the provider cannot enforce.
///
/// CHECK: `cargo test --lib runtime_policy_denials_fail_closed_before_spawn`
/// exits 0 (expected: fake launcher records zero spawn attempts and errors
/// classify the denied policy).
#[test]
fn ac_5_mik_6555_ac_5_ac_5_denied_network_and_mount_pol() {
    // ── Egress deny_default on local_compat → fail closed ───────────────
    let local = LocalCompatProvider::new();
    let mut config = RuntimeConfig::local_compat();
    config.egress.deny_default = true;

    let verdict = local.validate_policy(&config);
    assert!(!verdict.is_allowed(),
        "local_compat must deny egress.deny_default (cannot enforce)");
    assert!(verdict.denial_reason().unwrap().contains("local_compat"),
        "denial reason must name the provider");

    // ── Docker socket mount → fail closed ───────────────────────────────
    let docker = DockerProvider::new("docker");
    for sock_path in FORBIDDEN_DOCKER_SOCKET_PATHS {
        let mut sock_config = RuntimeConfig::docker_restricted();
        sock_config.mounts.mounts.push(MountEntry {
            host: (*sock_path).to_string(),
            container: (*sock_path).to_string(),
            writable: false,
        });
        let v = docker.validate_policy(&sock_config);
        assert!(!v.is_allowed(),
            "docker socket mount '{sock_path}' must be denied");
    }

    // ── Forbidden mount paths → fail closed ─────────────────────────────
    for forbidden in FORBIDDEN_MOUNT_PATHS {
        let mut fconfig = RuntimeConfig::docker_restricted();
        fconfig.mounts.mounts.push(MountEntry {
            host: (*forbidden).to_string(),
            container: (*forbidden).to_string(),
            writable: false,
        });
        let v = docker.validate_policy(&fconfig);
        assert!(!v.is_allowed(),
            "mount of forbidden path '{forbidden}' must be denied");
    }

    // ── Relative paths → fail closed ────────────────────────────────────
    let rel_entry = MountEntry {
        host: "./relative/path".to_string(),
        container: "/container".to_string(),
        writable: false,
    };
    let policy = MountPolicy::default_restricted();
    let rel_verdict = validate_mount(&rel_entry, &policy);
    assert!(!rel_verdict.is_allowed(), "relative paths must be denied");

    let rel_no_slash = MountEntry {
        host: "relative/path".to_string(),
        container: "/container".to_string(),
        writable: false,
    };
    let rel2_verdict = validate_mount(&rel_no_slash, &policy);
    assert!(!rel2_verdict.is_allowed(),
        "non-absolute paths must be denied");

    // ── Path traversal → fail closed ────────────────────────────────────
    let traversal_host = MountEntry {
        host: "/etc/../passwd".to_string(),
        container: "/container".to_string(),
        writable: false,
    };
    let trav_verdict = validate_mount(&traversal_host, &policy);
    assert!(!trav_verdict.is_allowed(), "path traversal must be denied");

    let traversal_container = MountEntry {
        host: "/tmp/data".to_string(),
        container: "/etc/../secret".to_string(),
        writable: false,
    };
    let trav2_verdict = validate_mount(&traversal_container, &policy);
    assert!(!trav2_verdict.is_allowed(),
        "container path traversal must be denied");

    // ── Writable mount without allow_writable → fail closed ─────────────
    let writable_entry = MountEntry {
        host: "/tmp/data".to_string(),
        container: "/data".to_string(),
        writable: true,
    };
    let w_verdict = validate_mount(&writable_entry, &policy);
    assert!(!w_verdict.is_allowed(),
        "writable mount without allow_writable must be denied");

    // ── Egress allowlist that provider cannot enforce → fail closed ─────
    // local_compat cannot enforce egress; validate_egress rejects it
    let egress_policy = EgressPolicy {
        deny_default: true,
        allowlist: vec!["10.0.0.0/8".to_string()],
    };
    let e_verdict = validate_egress(&egress_policy, "local_compat");
    assert!(!e_verdict.is_allowed(),
        "egress deny_default on local_compat must be denied");

    // Docker CAN enforce egress, so it should be allowed
    let e2_verdict = validate_egress(&egress_policy, "docker");
    assert!(e2_verdict.is_allowed(),
        "docker should be able to enforce egress policy");
}
